from src.xiangqi.engine.legal import generate_legal_moves

def _perft(state, depth):
    if depth == 0: return 1
    moves = generate_legal_moves(state)
    if depth == 1: return len(moves)
    total = 0
    for m in moves:
        snap = state.copy()
        from src.xiangqi.engine.types import sq_to_rc
        fsq = m & 0x1FF
        tsq = (m >> 9) & 0x7F
        fr, fc = sq_to_rc(fsq); tr, tc = sq_to_rc(tsq)
        piece = int(snap.board[fr, fc])
        captured = int(snap.board[tr, tc])
        snap.board[tr, tc] = snap.board[fr, fc]
        snap.board[fr, fc] = 0
        if abs(piece) == 1: snap.king_positions[snap.turn] = tsq
        snap.turn *= -1
        total += _perft(snap, depth - 1)
    return total
