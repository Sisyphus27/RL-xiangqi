    Implement _handle_board_click, _is_legal_target, _execute_move, and apply_move:

    - `_is_legal_target(from_row, from_col, to_row, to_col) -> bool`:
      from_sq = rc_to_sq(from_row, from_col)
      to_sq = rc_to_sq(to_row, to_col)
      For each move in self._engine.legal_moves():
        move_from = move & 0x1FF
        move_to = (move >> 9) & 0x7F
        if move_from == from_sq and move_to == to_sq:
          return True
      return False

    - `_handle_board_click(row: int, col: int) -> None`:
      clicked_piece = self._piece_index.get((row, col))

      If self._selected is not None (has selection):
        from_row, from_col = self._selected
        if (row, col) == self._selected:
          self._deselect_piece()  # click same piece: deselect
        elif clicked_piece and clicked_piece._piece_value > 0:
          self._deselect_piece()
          self._select_piece(row, col)  # click another red piece: switch
        elif self._is_legal_target(from_row, from_col, row, col):
          self._execute_move(from_row, from_col, row, col)  # click legal target: move
        else:
          self._deselect_piece()  # click illegal target: deselect
      Else (no selection):
        if clicked_piece and clicked_piece._piece_value > 0:
          self._select_piece(row, col)  # click red piece: select
        # else: click empty or black piece: do nothing

    - `_execute_move(from_row, from_col, to_row, to_col) -> None`:
      from_sq = rc_to_sq(from_row, from_col)
      to_sq = rc_to_sq(to_row, to_col)
      self.apply_move(from_sq, to_sq)

    - `apply_move(from_sq: int, to_sq: int) -> None` (per D-10 to D-23):
      from_row, from_col = sq_to_rc(from_sq)
      to_row, to_col = sq_to_rc(to_sq)

      # Handle capture (D-16)
      captured_item = self._piece_index.get((to_row, to_col))
      if captured_item:
        self._scene.removeItem(captured_item)
        del self._piece_index[(to_row, to_col)]
        captured_val = captured_item._piece_value
      else:
        captured_val = 0

      # Apply to engine (D-11): board calls engine.apply
      move = encode_move(from_sq, to_sq, captured_val != 0)
      self._state = self._engine.apply(move)

      # Update piece item position (D-14): update _row/_col and setPos
      piece_item = self._piece_index[(from_row, from_col)]
      piece_item._row = to_row
      piece_item._col = to_col
      piece_item.setPos((to_col + 0.2) * self._cell, (to_row + 0.2) * self._cell)

      # Update index (D-17): delete old key, add new key
      del self._piece_index[(from_row, from_col)]
      self._piece_index[(to_row, to_col)] = piece_item

      # Clear highlights (D-13): _clear_highlights + _selected = None
      self._clear_highlights()
      self._selected = None

      # Emit signal (D-19, D-20): self.move_applied.emit(from_sq, to_sq, captured_val)

    - `set_interactive(enabled: bool) -> None` (per D-26, D-27):
      Set self._interactive = enabled
      If not enabled: call self._deselect_piece()
