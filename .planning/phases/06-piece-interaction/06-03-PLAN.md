```python
class QXiangqiBoard(QGraphicsView):
    # State variables
    _selected: tuple[int, int] | None        # (row, col) of selected piece
    _selection_ring: QGraphicsEllipseItem | None
    _highlight_items: list[QGraphicsEllipseItem]
    _piece_index: dict[tuple[int, int], PieceItem]
    _interactive: bool  # default True

    # Signal
    move_applied = pyqtSignal(int, int, int)  # from_sq, to_sq, captured

    # Methods available for testing
    set_interactive(enabled: bool) -> None
```
