       ```python
       """AIWorker - QThread worker object for AI computation."""
       from PyQt6.QtCore import QObject, pyqtSignal

       from src.xiangqi.ai.base import AIPlayer, EngineSnapshot


       class AIWorker(QObject):
           """Worker object for AI computation in a background thread.

           Per D-04 to D-06: Uses QThread + moveToThread() pattern.
           The worker is created on main thread, then moved to AI thread.
           compute() slot runs in worker thread; signals cross to main thread.

           Signals
           -------
           move_ready : int
               Emitted with 16-bit move integer when AI selects a move.
           error : str
               Emitted with error message if AI computation fails.
           """

           move_ready = pyqtSignal(int)
           error = pyqtSignal(str)

           def __init__(self, ai_player: AIPlayer, snapshot: EngineSnapshot):
               super().__init__()
               self._ai = ai_player
               self._snapshot = snapshot

           def compute(self) -> None:
               """Slot called when thread starts. Runs in worker thread.

               Calls AI suggest_move and emits result via signal.
               """
               try:
                   move = self._ai.suggest_move(self._snapshot)
                   if move is not None:
                       self.move_ready.emit(move)
               except Exception as e:
                   self.error.emit(str(e))
       ```

    - Snapshot was created on main thread before worker construction
    - compute() runs in worker thread, emits signal back to main thread

       _on_move_applied(from_sq, to_sq, captured):
       - Check engine.result(), if not IN_PROGRESS -> _handle_game_over
       - Emit turn_changed with engine.turn
       - If engine.turn == -1 (black): _start_ai_turn()

       _start_ai_turn():
       - Call board.set_interactive(False) per D-18
       - Emit ai_thinking_started
       - Create snapshot = EngineSnapshot.from_engine(engine)
       - Create QThread and AIWorker, moveToThread
       - Wire: thread.started -> worker.compute

       - Wire: worker.error -> _on_ai_error

       - Start thread

       - Quit/wait thread, clear refs
       - Emit ai_thinking_finished
       - Per D-11: if not engine.is_legal(move): raise ValueError
       - Decode move, call board.apply_move(from_sq, to_sq)
       - Call board.set_interactive(True)

       _on_ai_error(message):
       - Quit/wait thread, clear refs
       - Emit ai_thinking_finished
       - Call board.set_interactive(True)
       - Print error (don't crash)

       _handle_game_over(result):
       - Emit game_over with result

       _on_game_over(result):
       - Map result to Chinese: RED_WINS->红胜, BLACK_WINS->黑胜, DRAW->和棋
       - Show QMessageBox.information(window, "对局结束", result_text, Ok)

       - Get window.statusBar()
       - If ai_thinking: "AI 思考中..."
       - Elif turn == 1: "红方回合"
