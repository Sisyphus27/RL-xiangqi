**QXiangqiBoard class:**
```
class QXiangqiBoard(QGraphicsView):
    def __init__(self, state: XiangqiState | None = None, parent=None):
        super().__init__(parent)
        self._state = state or XiangqiState.starting()
        self._scene = QGraphicsScene(self)
        self.setScene(self._scene)
        self.setBackgroundRole(QPalette.ColorRole.NoRole)
        self.setHorizontalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.setVerticalScrollBarPolicy(Qt.ScrollBarPolicy.ScrollBarAlwaysOff)
        self.setFrameShape(QFrame.Shape.NoFrame)
        self._cell = 0.0
        # trigger initial layout
        self.resizeEvent(None)

    def resizeEvent(self, event: QResizeEvent | None):
        vp = self.viewport()
        vw, vh = vp.width(), vp.height()
        self._cell = min(vw, vh) / 11.2
        sw = 10.2 * self._cell
        sh = 11.2 * self._cell
        self._scene.setSceneRect(0, 0, sw, sh)
        # Remove old piece items
        for item in list(self._scene.items()):
            if isinstance(item, PieceItem):
                self._scene.removeItem(item)
        # Re-add all pieces
        self._load_pieces()
        self.viewport().update()

    def _load_pieces(self):
        board = self._state.board
        for row in range(ROWS):
            for col in range(COLS):
                val = int(board[row, col])
                if val == 0:
                    continue
                piece_item = PieceItem(row, col, val, self._cell)
                self._scene.addItem(piece_item)

    def drawBackground(self, painter: QPainter, rect: QRectF):
        cell = self._cell
        p = painter
        p.setRenderHint(QPainter.RenderHint.Antialiasing)

        # 1. Green felt background rectangle
        p.fillRect(-0.6*cell, -0.6*cell, 10.2*cell, 11.2*cell,
                   QBrush(QColor(BOARD_BG)))

        # 2. Grid lines
        p.setPen(QPen(QColor(GRID_COLOR), 1.0))

        # 3. Vertical lines (9 lines)
        for col in range(COLS):
            x = (col + 0.6) * cell
            p.drawLine(x, 0.6*cell, x, 10.6*cell)

        # 4. Horizontal lines (10 lines, skip river gap at row 4)
        for row in range(ROWS):
            if row == 4:  # skip line between row 4 and 5 (river gap)
                continue
            y = (row + 0.6) * cell
            p.drawLine(0.6*cell, y, 9.6*cell, y)

        # 5. Palace diagonals
        # Top palace: rows 0-2, cols 0-2 (left) and cols 6-8 (right)
        p.drawLine((0.6)*cell, (0.6)*cell, (2.6)*cell, (2.6)*cell)  # top-left \
        p.drawLine((2.6)*cell, (0.6)*cell, (0.6)*cell, (2.6)*cell)  # top-left /
        p.drawLine((6.6)*cell, (0.6)*cell, (8.6)*cell, (2.6)*cell)  # top-right \
        p.drawLine((8.6)*cell, (0.6)*cell, (6.6)*cell, (2.6)*cell)  # top-right /
        # Bottom palace: rows 7-9, cols 0-2 (left) and cols 6-8 (right)
        p.drawLine((0.6)*cell, (7.6)*cell, (2.6)*cell, (9.6)*cell)  # bottom-left \
        p.drawLine((2.6)*cell, (7.6)*cell, (0.6)*cell, (9.6)*cell)  # bottom-left /
        p.drawLine((6.6)*cell, (7.6)*cell, (8.6)*cell, (9.6)*cell)  # bottom-right \
        p.drawLine((8.6)*cell, (7.6)*cell, (6.6)*cell, (9.6)*cell)  # bottom-right /

        # 6. River text "楚河" / "漢界"
        font_river = QFont("SimSun, Microsoft YaHei, Arial")
        font_size_river = max(int(cell * RIVER_FONT_RATIO), 10)
        font_river.setPixelSize(font_size_river)
        p.setFont(font_river)
        p.setPen(QPen(QColor(RIVER_TEXT_COLOR)))
        p.drawText(QPointF(2.0*cell, 5.1*cell), "楚河")
        p.drawText(QPointF(5.5*cell, 5.1*cell), "漢界")

        # 7. Coordinate labels
        font_coord = QFont("SimSun, Microsoft YaHei, Arial")
        font_coord.setPixelSize(max(int(cell * COORD_FONT_RATIO), 8))
        p.setFont(font_coord)
        p.setPen(QPen(QColor(COORD_TEXT_COLOR)))
        # Column labels: 9 columns, numbered 1-9 left-to-right
        for col in range(COLS):
            x = (col + 0.6) * cell
            # Red perspective (bottom): 1-9 left-to-right
            p.drawText(QPointF(x - 4, 11.2*cell + 4), str(col + 1))
            # Black perspective (top): 9-1 right-to-left
            p.drawText(QPointF(x - 4, 0.4*cell), str(9 - col))
        # Row labels: 10 rows
        for row in range(ROWS):
            y = (row + 0.6) * cell
            # Red ranks (left): 1-10 bottom-to-top
            p.drawText(QPointF(0.05*cell, y + 4), str(10 - row))
            # Black ranks (right): 10-1 top-to-bottom
            p.drawText(QPointF(9.7*cell, y + 4), str(row + 1))
```

"""Tests for QXiangqiBoard — scene structure, piece loading, window title."""
import pytest
from src.xiangqi.engine.state import XiangqiState
from src.xiangqi.ui.board import QXiangqiBoard
from src.xiangqi.ui.board_items import PieceItem
from src.xiangqi.ui.constants import RED_FILL, BLACK_FILL

def test_board_scene_exists(board):
    assert board.scene() is not None

def test_piece_count(board):
    """Exactly 32 pieces on starting board."""
    pieces = [i for i in board.scene().items() if isinstance(i, PieceItem)]
    assert len(pieces) == 32

def test_red_pieces_count(board):
    """Red side: 16 pieces (7 types)."""
    pieces = [i for i in board.scene().items() if isinstance(i, PieceItem)]
    red_pieces = [p for p in pieces if p._piece_value > 0]
    assert len(red_pieces) == 16

def test_black_pieces_count(board):
    """Black side: 16 pieces (7 types)."""
    pieces = [i for i in board.scene().items() if isinstance(i, PieceItem)]
    black_pieces = [p for p in pieces if p._piece_value < 0]
    assert len(black_pieces) == 16

def test_piece_colors_match_constants(board):
    """Red fill is #CC2200, black fill is #1A1A1A."""
    for item in board.scene().items():
        if not isinstance(item, PieceItem):
            continue
        expected_fill = "#CC2200" if item._piece_value > 0 else "#1A1A1A"
        actual = item.brush().color().name().upper()
        assert actual == expected_fill.upper(), \
            f"Piece value {item._piece_value}: expected {expected_fill}, got {actual}"

def test_cell_size_formula():
    """cell = min(vw, vh) / 11.2 at various sizes."""
    test_cases = [
        (360, 450, 450 / 11.2),
        (480, 600, 600 / 11.2),
        (720, 900, 900 / 11.2),
        (400, 500, 400 / 11.2),  # width constrains
    ]
    for vw, vh, expected_cell in test_cases:
        cell = min(vw, vh) / 11.2
        assert abs(cell - expected_cell) < 0.01

def test_window_title_from_main():
    """MainWindow has correct title set."""
    from src.xiangqi.ui.main import MainWindow
    win = MainWindow()
    assert win.windowTitle() == "RL-Xiangqi v0.2"
