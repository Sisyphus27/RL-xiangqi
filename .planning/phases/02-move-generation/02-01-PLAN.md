```python
ROWS, COLS = 10, 9
NUM_SQUARES = 90

class Piece(IntEnum):
    EMPTY = 0
    R_SHUAI = +1; B_JIANG = -1   # general
    R_SHI = +2; B_SHI = -2        # advisor
    R_XIANG = +3; B_XIANG = -3   # elephant
    R_MA = +4; B_MA = -4          # horse
    R_CHE = +5; B_CHE = -5        # chariot
    R_PAO = +6; B_PAO = -6        # cannon
    R_BING = +7; B_ZU = -7        # soldier

def encode_move(from_sq: int, to_sq: int, is_capture: bool = False) -> int:
    return (from_sq & 0x1FF) | ((to_sq << 9) & 0xFE00) | (int(is_capture) << 16)

def decode_move(move: int) -> Tuple[int, int, bool]:
    from_sq = move & 0x1FF
    to_sq   = (move >> 9) & 0x7F
    capture = (move >> 16) & 0x1
    return from_sq, to_sq, bool(capture)

def rc_to_sq(row: int, col: int) -> int:
    return row * COLS + col   # 0-89

def sq_to_rc(sq: int) -> Tuple[int, int]:
    return divmod(sq, COLS)
```

**gen_general** -- orthogonal 1-step, palace-only. Use `IN_PALACE[nr, nc]` to constrain destinations. Source: MOVE-01, RULES.md §2.1.
```python
def gen_general(board: np.ndarray, from_sq: int, color: int) -> List[int]:
    """Orthogonal 1-step, confined to palace. Returns list of encoded moves."""
    fr, fc = sq_to_rc(from_sq)
    moves: List[int] = []
    for dr, dc in _ORTHOGONAL:
        nr, nc = fr + dr, fc + dc
        if not (0 <= nr < ROWS and 0 <= nc < COLS):
            continue
        if not IN_PALACE[nr, nc]:
            continue
        target = int(board[nr, nc])
        if not belongs_to(target, color):
            moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=target != 0))
    return moves
```

**gen_chariot** -- orthogonal slide, stop at any blocker (own or enemy). Source: MOVE-02, RULES.md §2.2.
```python
def gen_chariot(board: np.ndarray, from_sq: int, color: int) -> List[int]:
    """Orthogonal sliding, stop at any piece. Collect all reachable squares."""
    fr, fc = sq_to_rc(from_sq)
    moves: List[int] = []
    for dr, dc in _ORTHOGONAL:
        nr, nc = fr + dr, fc + dc
        while 0 <= nr < ROWS and 0 <= nc < COLS:
            target = int(board[nr, nc])
            if not belongs_to(target, color):
                moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=target != 0))
            if target != 0:  # stop on any piece (own or enemy)
                break
            nr += dr
            nc += dc
    return moves
```

**gen_horse** -- L-shape with leg-block check. Block the entire leg (both destinations) if the knee square is occupied. Source: MOVE-03, RULES.md §2.3, PITFALLS.md Pitfall 3.
```python
def gen_horse(board: np.ndarray, from_sq: int, color: int) -> List[int]:
    """L-shape: orthogonal 1-step (leg) then diagonal 1-step. Leg must be empty."""
    fr, fc = sq_to_rc(from_sq)
    moves: List[int] = []
    for leg_dr, leg_dc, (d1r, d1c), (d2r, d2c) in _HORSE_LEG_DEST:
        leg_r, leg_c = fr + leg_dr, fc + leg_dc
        if not (0 <= leg_r < ROWS and 0 <= leg_c < COLS):
            continue
        if board[leg_r, leg_c] != 0:
            continue  # leg is blocked -- skip both destinations
        for dest_dr, dest_dc in ((d1r, d1c), (d2r, d2c)):
            nr, nc = fr + dest_dr, fc + dest_dc
            if not (0 <= nr < ROWS and 0 <= nc < COLS):
                continue
            target = int(board[nr, nc])
            if not belongs_to(target, color):
                moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=target != 0))
    return moves
```

**gen_cannon** -- slide non-capture, exactly-1-screen capture. Use per-direction screen_count state machine. CRITICAL: cannon CAN capture own piece as screen (any color), and CANNOT capture with 0 screens or 2+ screens. Source: MOVE-04, RULES.md §2.4, PITFALLS.md Pitfall 6.
```python
def gen_cannon(board: np.ndarray, from_sq: int, color: int) -> List[int]:
    """Orthogonal sliding: non-capture slides, capture requires exactly 1 screen."""
    fr, fc = sq_to_rc(from_sq)
    moves: List[int] = []
    for dr, dc in _ORTHOGONAL:
        nr, nc = fr + dr, fc + dc
        screen_count = 0
        while 0 <= nr < ROWS and 0 <= nc < COLS:
            target = int(board[nr, nc])
            if screen_count == 0:
                # Sliding mode: collect empty squares, stop at first piece
                if target == 0:
                    moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=False))
                else:
                    screen_count = 1  # first piece found -- switch to capture mode
            else:
                # Capture mode: screen_count == 1
                if target == 0:
                    # Empty square after screen -- keep searching
                    pass
                else:
                    # Second piece encountered -- if enemy, it is capturable
                    if not belongs_to(target, color):
                        moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=True))
                    break  # stop after second piece (regardless of capture or not)
            nr += dr
            nc += dc
    return moves
```

**gen_elephant** -- diagonal 2-step, eye-block (midpoint empty), river constraint (red: row 0-4, black: row 5-9). Both constraints independently checked. Source: MOVE-06, RULES.md §2.6, PITFALLS.md Pitfall 4.
```python
def gen_elephant(board: np.ndarray, from_sq: int, color: int) -> List[int]:
    """Diagonal 2-step: eye midpoint must be empty; cannot cross river."""
    fr, fc = sq_to_rc(from_sq)
    # Red elephant home half: rows 0-4; black elephant home half: rows 5-9
    home_rows = range(0, 5) if color == +1 else range(5, 10)
    moves: List[int] = []
    for dr, dc, er, ec in _ELEPHANT_EYE_DEST:
        nr, nc = fr + dr, fc + dc
        eye_r, eye_c = fr + er, fc + ec
        if not (0 <= nr < ROWS and 0 <= nc < COLS):
            continue
        if nr not in home_rows:
            continue  # river constraint: cannot cross
        if board[eye_r, eye_c] != 0:
            continue  # elephant eye blocked
        target = int(board[nr, nc])
        if not belongs_to(target, color):
            moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=target != 0))
    return moves
```

**gen_soldier** -- forward 1-step always; forward+sideways after crossing river. Red sideways enabled when fr >= 5 (row 5+ is black's side). Black sideways enabled when fr <= 4. Source: MOVE-07, RULES.md §2.7, PITFALLS.md (soldier movement notes).
```python
def gen_soldier(board: np.ndarray, from_sq: int, color: int) -> List[int]:
    """Forward 1-step always; forward+sideways after crossing river."""
    fr, fc = sq_to_rc(from_sq)
    moves: List[int] = []
    forward_dr = +1 if color == +1 else -1
    # Forward move
    nr, nc = fr + forward_dr, fc
    if 0 <= nr < ROWS:
        target = int(board[nr, nc])
        if not belongs_to(target, color):
            moves.append(encode_move(from_sq, rc_to_sq(nr, nc), is_capture=target != 0))
    # Sideways moves only after crossing river
    # Red: rows 5-9 are black's side; black: rows 0-4 are red's side
    crossed = (color == +1 and fr >= 5) or (color == -1 and fr <= 4)
    if crossed:
        for dc in (-1, +1):
            nc_s = fc + dc
            if 0 <= nc_s < COLS:
                target = int(board[fr, nc_s])
                if not belongs_to(target, color):
                    moves.append(encode_move(from_sq, rc_to_sq(fr, nc_s), is_capture=target != 0))
    return moves
```

**is_in_check(state, color)** -- O(1) king lookup via king_positions. Scan all attack sources. MUST include face-to-face as a check source (RULES.md §9.15). Source: RULE-03, PITFALLS.md Pitfall 5.

Implementation approach:
1. Get king position from `state.king_positions[color]` -- O(1). Return False if king not found.
2. Convert to row/col: `kr, kc = divmod(ksq, 9)`.
3. Scan 4 orthogonal rays for chariot (first piece), cannon (exactly 1 screen), and face-to-face general.
4. Scan horse destinations (leg must be empty).
5. Scan soldier forward square (only same column, adjacent row; sideways only if soldier across river).
6. Scan advisor diagonal adjacent within palace.
7. Scan elephant diagonal adjacent on own side (eye empty).
8. Return True if any attacker found.

The soldier attack logic in is_in_check MUST be: check the square directly in front of the king (same column, adjacent row). If that square has an enemy soldier, the king is in check. Sideways soldier attack: only if the soldier is across the river. (RULES.md §2.7: soldiers attack orthogonally forward only -- no diagonal attack.)

```python
def is_in_check(state: XiangqiState, color: int) -> bool:
    """Return True if the general of given color is under attack."""
    ksq = state.king_positions.get(color)
    if ksq is None:
        return False
    kr, kc = divmod(ksq, 9)
    enemy = -color

    # 1. Orthogonal attackers: chariot, face-to-face general, cannon
    for dr, dc in _ORTHOGONAL:
        nr, nc = kr + dr, kc + dc
        first_piece_sq = None
        first_piece_val = 0
        screen_count = 0
        second_piece_sq = None
        second_piece_val = 0
        while 0 <= nr < ROWS and 0 <= nc < COLS:
            p = int(state.board[nr, nc])
            if p != 0:
                if screen_count == 0:
                    first_piece_sq = rc_to_sq(nr, nc)
                    first_piece_val = p
                    screen_count = 1
                else:
                    second_piece_sq = rc_to_sq(nr, nc)
                    second_piece_val = p
                    break
            nr += dr
            nc += dc
        if first_piece_sq is not None:
            if first_piece_val == enemy * 5:  # chariot attacks on same line
                return True
            if screen_count == 1 and second_piece_sq is not None:
                if second_piece_val == enemy * 6:  # cannon with exactly 1 screen
                    return True
                if first_piece_val == enemy * 1:  # face-to-face general (no screen)
                    return True

    # 2. Horse (L-shape, leg must be empty)
    for leg_dr, leg_dc, (d1r, d1c), (d2r, d2c) in _HORSE_LEG_DEST:
        leg_r, leg_c = kr + leg_dr, kc + leg_dc
        if not (0 <= leg_r < ROWS and 0 <= leg_c < COLS):
            continue
        if state.board[leg_r, leg_c] != 0:
            continue
        for dest_dr, dest_dc in ((d1r, d1c), (d2r, d2c)):
            nr, nc = kr + dest_dr, kc + dest_dc
            if not (0 <= nr < ROWS and 0 <= nc < COLS):
                continue
            if int(state.board[nr, nc]) == enemy * 4:  # horse
                return True

    # 3. Soldier: forward square only (adjacent row, same col)
    srow = kr + (1 if color == +1 else -1)
    if 0 <= srow < ROWS:
        if int(state.board[srow, kc]) == enemy * 7:  # soldier forward
            return True
    # Sideways: only if soldier is across the river
    for dc in (-1, +1):
        sc = kc + dc
        if 0 <= sc < COLS:
            soldier_row = srow
            if 0 <= soldier_row < ROWS:
                p = int(state.board[soldier_row, sc])
                if p == enemy * 7:
                    # Soldier attacks sideways only if it is across the river
                    if (enemy == +1 and soldier_row >= 5) or (enemy == -1 and soldier_row <= 4):
                        return True

    # 4. Advisor: adjacent diagonal within palace
    for dr, dc in _DIAGONAL:
        nr, nc = kr + dr, kc + dc
        if 0 <= nr < ROWS and 0 <= nc < COLS:
            if IN_PALACE[nr, nc] and int(state.board[nr, nc]) == enemy * 2:
                return True

    # 5. Elephant: adjacent diagonal on own half (eye must be empty)
    for dr, dc in _DIAGONAL:
        nr, nc = kr + dr, kc + dc
        er, ec = kr + 2*dr, kc + 2*dc
        if 0 <= nr < ROWS and 0 <= nc < COLS and 0 <= er < ROWS and 0 <= ec < COLS:
            if int(state.board[nr, nc]) == 0:  # eye empty
                if int(state.board[er, ec]) == enemy * 3:
                    return True

    return False
```

**apply_move(state, move)** -- updates board, king_positions, turn, zobrist hash, halfmove_clock, move_history. Returns captured piece (0 if none). Source: RESEARCH.md §Apply/Unmake.
```python
def apply_move(state: XiangqiState, move: int) -> int:
    """Apply move to state. Returns captured piece (0 if none)."""
    from_sq, to_sq, _ = decode_move(move)
    fr, fc = sq_to_rc(from_sq)
    tr, tc = sq_to_rc(to_sq)
    piece = int(state.board[fr, fc])
    captured = int(state.board[tr, tc])

    # Update board
    state.board[tr, tc] = np.int8(piece)
    state.board[fr, fc] = np.int8(0)

    # Update king_positions if general moved
    pt = abs(piece)
    if pt == 1:  # general
        state.king_positions[state.turn] = to_sq

    # Update turn
    state.turn *= -1

    # Update Zobrist hash
    from .state import update_hash
    new_hash = update_hash(
        state.zobrist_hash_history[-1],
        from_sq, to_sq, piece, captured, state.turn
    )
    state.zobrist_hash_history.append(new_hash)

    # Update history and halfmove_clock
    state.move_history.append(move)
    if pt == 7 or captured != 0:
        state.halfmove_clock = 0
    else:
        state.halfmove_clock += 1

    return captured
```

```python
def is_legal_move(state: XiangqiState, move: int) -> bool:
    """Return True if move is legal for the side to move.

    Legality = pseudo-legal + own general not left in check + no flying general violation.
    """
    snap = state.copy()
    apply_move(snap, move)
    # Own king (color to move) must be safe after the move
    if is_in_check(snap, state.turn):
        return False
    # No flying general violation
    if flying_general_violation(snap, state.turn):
        return False
    return True
```

```python
"""Xiangqi rule helpers: flying general detection, game result evaluation."""
from __future__ import annotations

import numpy as np

from .types import ROWS, COLS, rc_to_sq, sq_to_rc


def _generals_face_each_other(board: np.ndarray, king_positions: dict[int, int]) -> int | None:
    """Return the column index if both generals face each other (same col, nothing between), else None."""
    rk = king_positions.get(+1)
    bk = king_positions.get(-1)
    if rk is None or bk is None:
        return None
    rr, rc = divmod(rk, 9)
    br, bc = divmod(bk, 9)
    if rc != bc:
        return None
    lo, hi = min(rr, br), max(rr, br)
    for r in range(lo + 1, hi):
        if board[r, rc] != 0:
            return None
    return rc  # they face each other on this file


def flying_general_violation(state_after_move, turn_before_move: int) -> bool:
    """Return True if both generals would face each other after a move is applied.

    This is a violation: no piece may move to clear a file between the two generals.
    """
    col = _generals_face_each_other(state_after_move.board, state_after_move.king_positions)
    return col is not None


def get_game_result(state) -> str:
    """Return game result: 'RED_WINS', 'BLACK_WINS', 'DRAW', or 'IN_PROGRESS'.

    Checks in order of priority:
    1. No legal moves + in check -> checkmate (current player loses)
    2. No legal moves + not in check -> stalemate (困毙) = loss in Xiangqi (current player loses)
    3. Otherwise -> IN_PROGRESS
    """
    from .legal import generate_legal_moves, is_in_check

    legal = generate_legal_moves(state)
    if len(legal) > 0:
        return 'IN_PROGRESS'

    # No legal moves -- check if in check
    in_check = is_in_check(state, state.turn)
    if in_check:
        # Checkmate: current player loses
        return 'BLACK_WINS' if state.turn == +1 else 'RED_WINS'
    else:
        # Stalemate (困毙) = loss in Xiangqi (not draw)
        return 'BLACK_WINS' if state.turn == +1 else 'RED_WINS'


__all__ = ['flying_general_violation', '_generals_face_each_other', 'get_game_result']
```
