<objective>
Fix two correctness bugs and one test expectation error in the Phase 4 API engine.

Bug 1 (MAJOR): `is_legal()` (via `is_legal_move()`) returns True for geometrically illegal moves (e.g., diagonal chariot moves, own-piece captures). Root cause: `is_legal_move()` only validates own-piece-on-source and post-move in-check/flying-general, not move geometry per piece type. Fix: add piece-type geometry validation BEFORE the board copy.

Bug 2: `legal_moves()` averages 29ms vs 10ms target. Root cause: `is_legal_move()` copies the full board for every pseudo-legal candidate. Fix: pre-filter own-piece destinations in `generate_legal_moves()` before calling `is_legal_move()` to skip the expensive board copy.

Bug 3: `test_undo_empty_stack_raises` expects bare `IndexError` but `engine.undo()` raises `IndexError("nothing to undo")` per spec. No code change — update the test's `match=` argument.
</objective>

```python
# ─── Piece-geometry validators ─────────────────────────────────────────────

def _is_valid_geometry(piece: int, fr: int, fc: int, tr: int, tc: int, board: np.ndarray) -> bool:
    """Return True if (fr,fc)->(tr,tc) is a geometrically valid move for piece type.

    piece is the board cell value (positive for red, negative for black).
    This check is piece-type-specific; it does NOT check board occupancy.
    """
    pt = abs(piece)
    dr, dc = tr - fr, tc - fc
    color = +1 if piece > 0 else -1

    if pt == 1:  # General: orthogonal 1-step within palace
        if (abs(dr), abs(dc)) == (0, 1) or (abs(dr), abs(dc)) == (1, 0):
            return bool(IN_PALACE[tr, tc])
        return False

    if pt == 2:  # Advisor: diagonal 1-step within palace
        if (abs(dr), abs(dc)) == (1, 1):
            return bool(IN_PALACE[tr, tc])
        return False

    if pt == 3:  # Elephant: diagonal 2-step, eye must be empty, cannot cross river
        if (abs(dr), abs(dc)) != (2, 2):
            return False
        eye_r, eye_c = fr + dr // 2, fc + dc // 2
        home_rows = range(5, 10) if color == +1 else range(0, 5)
        if (tr not in home_rows) or (board[eye_r, eye_c] != 0):
            return False
        return True

    if pt == 4:  # Horse: L-shape (orthogonal 1-step leg + diagonal 1-step)
        for leg_dr, leg_dc, d1, d2 in _HORSE_LEG_DEST:
            leg_r, leg_c = fr + leg_dr, fc + leg_dc
            if not (0 <= leg_r < ROWS and 0 <= leg_c < COLS):
                continue
            for dest_dr, dest_dc in d1, d2:
                if tr - fr == dest_dr and tc - fc == dest_dc:
                    # Leg must be empty (board check is done by caller for efficiency)
                    return True
        return False

    if pt == 5:  # Chariot: orthogonal sliding
        if dr != 0 and dc != 0:
            return False
        step_r = 0 if dr == 0 else (1 if dr > 0 else -1)
        step_c = 0 if dc == 0 else (1 if dc > 0 else -1)
        nr, nc = fr + step_r, fc + step_c
        while (nr, nc) != (tr, tc):
            if board[nr, nc] != 0:
                return False
            nr += step_r
            nc += step_c
        return True

    if pt == 6:  # Cannon: orthogonal sliding, capture needs exactly 1 screen
        if dr != 0 and dc != 0:
            return False
        step_r = 0 if dr == 0 else (1 if dr > 0 else -1)
        step_c = 0 if dc == 0 else (1 if dc > 0 else -1)
        nr, nc = fr + step_r, fc + step_c
        screens = 0
        while (nr, nc) != (tr, tc):
            if board[nr, nc] != 0:
                screens += 1
                if screens > 1:
                    return False
            nr += step_r
            nc += step_c
        # Destination reached; no occupancy check here (done by caller)
        return True

    if pt == 7:  # Soldier: forward 1-step; sideways only after crossing river
        forward_dr = -1 if color == +1 else +1
        if dr == forward_dr and dc == 0:
            return True
        crossed = (color == +1 and fr <= 4) or (color == -1 and fr >= 5)
        if crossed and dr == 0 and (dc == -1 or dc == +1):
            return True
        return False

    return False  # unknown piece type
```

```python
def generate_legal_moves(state: XiangqiState) -> List[int]:
    """Return all legal moves for the side to move."""
    moves: List[int] = []
    for piece_sq in all_pieces_of_color(state.board, state.turn):
        fr, fc = sq_to_rc(piece_sq)
        piece = int(state.board[fr, fc])
        pt = abs(piece)
        if pt == 1:
            candidates = gen_general(state.board, piece_sq, state.turn)
        elif pt == 2:
            candidates = gen_advisor(state.board, piece_sq, state.turn)
        elif pt == 3:
            candidates = gen_elephant(state.board, piece_sq, state.turn)
        elif pt == 4:
            candidates = gen_horse(state.board, piece_sq, state.turn)
        elif pt == 5:
            candidates = gen_chariot(state.board, piece_sq, state.turn)
        elif pt == 6:
            candidates = gen_cannon(state.board, piece_sq, state.turn)
        elif pt == 7:
            candidates = gen_soldier(state.board, piece_sq, state.turn)
        else:
            candidates = []
        # Pre-filter: skip is_legal_move (and its board copy) for own-piece destinations
        for m in candidates:
            to_sq = (m >> 9) & 0x7F
            tr, tc = sq_to_rc(to_sq)
            dest_piece = int(state.board[tr, tc])
            if dest_piece != 0 and (dest_piece > 0) == (piece > 0):
                continue  # own piece blocks — geometrically invalid, skip board copy
            if is_legal_move(state, m):
                moves.append(m)
    return moves
```

4. Skip `is_legal_move()` if `dest_piece` belongs to the same color as `piece` (own-piece block)
