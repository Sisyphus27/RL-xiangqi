    ```python
    def step(self, action):
        """
        Execute one half-move.

        Returns (obs, reward, terminated, truncated, info).
        - action: flat integer 0-8099 (from_sq * 90 + to_sq)
        - terminated=True when game ends (win/loss/draw)
        - truncated=False (no step limit in Xiangqi)
        - reward: from red's perspective (+1= red wins, -1= red loses, 0=draw)
        """
        # D-06: Illegal move handling first
        from_sq = action // 90   # D-03: decode flat action
        to_sq = action % 90

        # Convert flat action to engine's 16-bit move encoding
        from xiangqi.engine.types import encode_move
        move = encode_move(from_sq, to_sq)

        if not self._engine.is_legal(move):
            # D-06: Illegal move penalty, state unchanged
            return (
                self._get_observation(),
                -2.0,
                False,
                False,
                {**self._get_info(), "illegal_move": True},
            )

        # Legal move: apply and compute reward
        captured = self._engine.apply(move)

        # Compute reward (D-07 terminal detection, R5 material reward)
        reward = self._compute_reward(captured)

        # Check terminal state
        result = self._engine.result()
        terminated = (result != "IN_PROGRESS")

        if terminated:
            reward = self._terminal_reward(result)

        obs = self._get_observation()
        info = self._get_info()
        info["illegal_move"] = False

        # truncated is always False for Xiangqi (no step limit)
        return obs, reward, terminated, False, info

    def _compute_reward(self, captured: int) -> float:
        """
        Compute non-terminal reward for the acting player (from red's perspective).
        R5: material capture = +/- piece_value / 100

        captured: piece value from engine.apply() return (0 if no capture)
        """
        if captured == 0:
            return 0.0

        piece_value = self.PIECE_VALUES.get(abs(captured), 0)
        # Red captured (engine turn was red) = positive reward
        # Black captured (engine turn was black) = negative reward
        sign = 1 if self._engine.turn == -1 else -1  # turn was just switched by apply()
        # Actually: self._engine.turn AFTER apply() is the NEXT player to move
        # The player who JUST moved is -self._engine.turn
        acting_player = -self._engine.turn
        sign = 1 if acting_player == 1 else -1  # red = +1, black = -1
        return sign * piece_value / 100.0

    def _terminal_reward(self, result: str) -> float:
        """
        Map engine result to terminal reward from red's perspective.
        R5: +1.0 (red wins), -1.0 (black wins), 0.0 (draw)
        """
        if result == "RED_WINS":
            return 1.0
        elif result == "BLACK_WINS":
            return -1.0
        else:  # DRAW
            return 0.0
    ```

    CRITICAL IMPLEMENTATION NOTES:

    1. **Action decoding (D-03)**: action is flat integer 0-8099 = from_sq * 90 + to_sq
       - `from_sq = action // 90` (0-89)
       - `to_sq = action % 90` (0-89)
       - Convert to engine 16-bit: `move = encode_move(from_sq, to_sq)`

    2. **Illegal move handling (D-06)**:
       - Return reward=-2.0, terminated=False, truncated=False
       - State is NOT updated (do NOT call engine.apply())
       - Set info["illegal_move"]=True

    3. **Legal move flow**:
       - Call engine.apply(move) to get captured piece
       - Compute material reward (R5: +/- piece_value/100)
       - Check engine.result() for terminal (D-07)
       - If terminal: override reward with terminal reward (R5)

    4. **Reward perspective**:
       - Reward is from RED's perspective: +1 = red wins, -1 = black wins
       - Material reward: red captures = positive, black captures = negative
       - Acting player before turn switch = -self._engine.turn (after apply())

    5. **Canonical rotation in step**:
       - The canonical board is maintained by _canonical_board() and used in _get_observation()
       - piece_type_to_move is computed from canonical board
       - When building masks in _build_piece_masks(), use canonical coordinates

    6. **DO NOT mutate info dict** - create a new one each time

    ```python
    def test_sync_vector_env():
        """R8: SyncVectorEnv with n_envs=2 produces valid episodes."""
        import gymnasium as gym
        def make_env(rank):
            def _init():
                env = gym.make("Xiangqi-v0")
                env.reset(seed=rank)
                return env
            return _init
        vec_env = gym.vector.SyncVectorEnv([make_env(0), make_env(1)])
        obs, infos = vec_env.reset()
        assert obs.shape == (2, 16, 10, 9), f"vec env reset obs shape {obs.shape}"

        # Step several times to verify no crashes
        for _ in range(10):
            # Get legal actions for both envs
    ```
