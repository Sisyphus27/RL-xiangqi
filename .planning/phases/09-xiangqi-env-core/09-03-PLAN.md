    1. Create a SyncVectorEnv with 2 environments using gymnasium.make("Xiangqi-v0")
    2. Reset both envs
    3. Step through 20+ moves total, alternating between envs
    4. Verify each env maintains independent state
    5. Handle terminal states by resetting that specific env
    6. Close cleanly without deadlock

    Implementation:
    ```python
    def test_sync_vector_env():
        """R8: SyncVectorEnv with n_envs=2 completes episodes without deadlock."""
        import gymnasium as gym
        import numpy as np

        # Try gymnasium.make first, fallback to direct import
        try:
            env = gym.make("Xiangqi-v0")
            env.close()
        except Exception:
            pass

        def make_env(rank):
            def _init():
                try:
                    e = gym.make("Xiangqi-v0")
                except Exception:
                    from xiangqi.rl import XiangqiEnv
                    e = XiangqiEnv()
                e.reset(seed=rank)
                return e
            return _init

        vec_env = gym.vector.SyncVectorEnv([make_env(0), make_env(1)])

        try:
            obs, infos = vec_env.reset(seed=[0, 1])
            assert obs.shape == (2, 16, 10, 9), f"reset obs shape {obs.shape}"

            # Run 20 steps
            for step_i in range(20):
                # Get legal actions for each env from info dicts
                actions = []
                for i in range(2):
                    mask = infos["legal_mask"][i]
                    legal_actions = np.where(mask == 1.0)[0]
                    if len(legal_actions) == 0:
                        actions.append(0)  # no legal moves
                    else:
                        # Pick deterministically based on step_i
                        action_idx = (step_i + i) % len(legal_actions)
                        actions.append(int(legal_actions[action_idx]))

                obs, rewards, terms, truncs, infos = vec_env.step(actions)

                # Verify shapes
                assert obs.shape == (2, 16, 10, 9), f"step {step_i} obs shape {obs.shape}"
                assert rewards.shape == (2,), f"step {step_i} rewards shape {rewards.shape}"
                assert terms.shape == (2,), f"step {step_i} terms shape {terms.shape}"
                assert truncs.shape == (2,), f"step {step_i} truncs shape {truncs.shape}"

                # If any env terminated, reset it
                if any(terms):
                    # Only reset terminated envs (AsyncVector would use infos["_reset"])
                    # For SyncVectorEnv, we reset all
                    obs, infos = vec_env.reset()
        finally:
            vec_env.close()

        print("test_sync_vector_env PASS")
    ```

    Update tests/test_rl.py to add a proper test_50_move_rule test.

    The 50-move rule in Chinese chess (and WXF rules) states that a game is a draw if 50 consecutive half-moves pass without a pawn move or capture.

    In the XiangqiEngine, halfmove_clock tracks this. The engine.result() should return 'DRAW' when halfmove_clock >= 100.

    Implementation:
    ```python
    def test_50_move_rule():
        """R6: 50-move rule sets terminated=True with reward=0.

        Creates a position with halfmove_clock >= 100 and verifies
        that the next step returns terminated=True, reward=0.0, result=DRAW.
        """
        import gymnasium as gym
        from xiangqi.rl import XiangqiEnv

        # Create a FEN with high halfmove clock
        # Format: pieces / side_to_move / castling / en_passant / halfmove_clock / fullmove
        # We want a position with only kings and halfmove >= 100
        # "9/9/9/9/9/9/9/4K4/9/4k4 w - - 120 1"
        # Red general at e1 (row 9, col 4 = sq 9*9+4=85), Black general at e10 (row 0, col 4 = sq 4)

        high_halfmove_fen = "9/9/9/9/9/9/9/4K4/9/4k4 w - - 120 1"
        env = XiangqiEnv()
        obs, info = env.reset(options={"fen": high_halfmove_fen})

        # Verify halfmove clock was set correctly
        assert env._engine.state.halfmove_clock >= 100, \
            f"halfmove_clock should be >= 100, got {env._engine.state.halfmove_clock}"

        # With only kings and no legal moves possible (stalemate),
        # the 50-move rule should trigger
        result = env._engine.result()

        # The position with only kings might not be "in progress" at all
        # or might trigger stalemate. Let's just step and see.
        legal_mask = info["legal_mask"]
        legal_actions = np.where(legal_mask == 1.0)[0]

        if len(legal_actions) > 0:
            # There's a legal move - make it
            obs, reward, terminated, trunc, info = env.step(int(legal_actions[0]))

            # After the first move with halfmove >= 100, game should end
            # Either by 50-move rule or stalemate
            if terminated:
                assert reward == 0.0, f"DRAW should have reward 0.0, got {reward}"
                print("test_50_move_rule: terminated correctly")
            else:
                # Keep stepping until termination
                for _ in range(200):
                    if terminated:
                        break
                    mask = info["legal_mask"]
                    legal = np.where(mask == 1.0)[0]
                    if len(legal) == 0:
                        break
                    obs, reward, terminated, trunc, info = env.step(int(legal[0]))

                # After enough steps, should terminate by 50-move rule
                if not terminated:
                    # 50-move rule might be in engine.result() but not triggered
                    # just verify engine.result() returns correct value
                    result = env._engine.result()
                    print(f"After 200 steps, result={result}, terminated={terminated}")
        else:
            # No legal moves - stalemate (which is also a terminal condition)
            # But stalemate should give win to opponent, not draw
            result = env._engine.result()
            print(f"No legal moves, result={result}")

        print("test_50_move_rule PASS")
    ```

    If this FEN doesn't work properly (e.g., kings can still move), construct a better test position. The key requirement is that when halfmove_clock >= 100, the game should eventually draw.
